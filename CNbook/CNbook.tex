\begin{algorithm}[!h]
	\caption{PARTITION$(A,p,r)$}%算法标题
	\textbf{Input: }{$\mathbf{x},
    \mathbf{y}\in\mathbb{R}^n$}\\
    \textbf{Output:}
    {$\mathbf{z}=\mathbf{x}
    \mathbf{y}^T\in\mathbb{R}^{n\times n}$}
    \begin{algorithmic}[1]%一行一个标行号
		\STATE $i=p$
		\FOR{$j=p$ to $r$}
		\IF{$A[j]<=0$}
		\STATE $swap(A[i],A[j])$
		\STATE $i=i+1$
		\ENDIF
		\ENDFOR
    \end{algorithmic}
\end{algorithm}
